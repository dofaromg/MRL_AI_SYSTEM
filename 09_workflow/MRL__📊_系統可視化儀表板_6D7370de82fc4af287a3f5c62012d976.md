## 2. 多維宇宙地圖

### 11 維空間可視化

```tsx
// origin_signature="MrLiouWord"

interface UniverseMapVisualization {
  universes: UniverseNode[];
  connections: UniverseConnection[];
  projection_mode: 'pca' | 't-sne' | 'umap' | '3d-projection';
  
  rendering: {
    show_coordinates: boolean;
    show_entanglement: boolean;
    show_labels: boolean;
    color_by: 'type' | 'health' | 'activity' | 'coherence';
  };
}

interface UniverseNode {
  id: string;
  name: string;
  connector_type: string;
  
  // 11 維座標
  coordinates: number[11];
  
  // 投影到 3D 的座標（用於渲染）
  projected_coordinates: [number, number, number];
  
  // 視覺屬性
  visual: {
    size: number;  // 基於活動量
    color: string;  // 基於類型或狀態
    opacity: number;  // 基於一致性
    glow: boolean;  // 是否發光（活躍狀態）
  };
  
  // 狀態信息
  status: {
    health: 'healthy' | 'degraded' | 'offline';
    coherence: number;
    activity_score: number;
    last_sync: string;
  };
  
  metadata: {
    origin_signature: 'MrLiouWord';
  };
}

interface UniverseConnection {
  source_id: string;
  target_id: string;
  
  // 連接類型
  type: 'entanglement' | 'sync' | 'data-flow';
  
  // 連接強度（影響線條粗細）
  strength: number;  // 0-1
  
  // 數據流量（影響動畫速度）
  flow_rate: number;
  
  // 視覺屬性
  visual: {
    color: string;
    width: number;
    animated: boolean;
  };
}

/**
 * 渲染 3D 宇宙地圖
 */
function renderUniverseMap(
  data: UniverseMapVisualization
): JSX.Element {
  return (
    <Canvas className="universe-map-canvas">
      {/* 場景設置 */}
      <Scene>
        <ambientLight intensity={0.5} />
        <pointLight position={[10, 10, 10]} />
        
        {/* 渲染宇宙節點 */}
        {data.universes.map(universe => (
          <UniverseSphere
            key={universe.id}
            position={universe.projected_coordinates}
            size={universe.visual.size}
            color={universe.visual.color}
            opacity={universe.visual.opacity}
            glow={universe.visual.glow}
            onClick={() => selectUniverse(universe.id)}
          />
        ))}
        
        {/* 渲染連接線 */}
        {data.connections.map((conn, idx) => (
          <ConnectionLine
            key={idx}
            start={getUniversePosition(conn.source_id)}
            end={getUniversePosition(conn.target_id)}
            color={conn.visual.color}
            width={conn.visual.width}
            animated={conn.visual.animated}
            flowRate={conn.flow_rate}
          />
        ))}
        
        {/* 座標軸（如果啟用）*/}
        {data.rendering.show_coordinates && (
          <CoordinateAxes dimensions={3} />
        )}
      </Scene>
      
      {/* 相機控制 */}
      <OrbitControls
        enableZoom={true}
        enablePan={true}
        enableRotate={true}
      />
    </Canvas>
  );
}
```

interface MemoryPoolVisualization {
  // 記憶統計
  stats: {
    total_memories: number;
    by_type: Record<string, number>;
    by_universe: Record<string, number>;
    by_importance: {
      high: number;
      medium: number;
      low: number;
    };
  };
  
  // 存儲使用情況
  storage: {
    total_capacity_gb: number;
    used_gb: number;
    available_gb: number;
    usage_percentage: number;
  };
  
  // 性能指標
  performance: {
    avg_retrieval_time_ms: number;
    cache_hit_rate: number;
    sync_operations_per_sec: number;
    failed_operations_count: number;
  };
  
  // 最近記憶
  recent_memories: MemoryNode[];
}

interface MemoryNode {
  id: string;
  type: 'conversation' | 'knowledge' | 'context' | 'pattern';
  content_preview: string;
  importance: number;
  created_at: string;
  universes: string[];
  size_kb: number;
  
  metadata: {
    origin_signature: 'MrLiouWord';
  };
}

/**
 * 記憶池儀表板組件
 */
function MemoryPoolDashboard() {
  const memoryData = useMemoryPoolData();
  
  return (
    <div className="memory-pool-dashboard">
      {/* 存儲使用圓環圖 */}
      <Card title="存儲使用情況">
        <DonutChart
          data={[
            { label: '已使用', value: memoryData.storage.used_gb, color: '#4F46E5' },
            { label: '可用', value: memoryData.storage.available_gb, color: '#E5E7EB' }
          ]}
          centerText={`${memoryData.storage.usage_percentage.toFixed(1)}%`}
        />
        <MetricRow>
          <Metric label="已使用" value={`${memoryData.storage.used_gb.toFixed(2)} GB`} />
          <Metric label="總容量" value={`${memoryData.storage.total_capacity_gb.toFixed(2)} GB`} />
        </MetricRow>
      </Card>
      
      {/* 記憶類型分佈 */}
      <Card title="記憶類型分佈">
        <BarChart
          data={Object.entries(memoryData.stats.by_type).map(([type, count]) => ({
            label: type,
            value: count
          }))}
          colorScheme="blue"
        />
      </Card>
      
      {/* 性能指標 */}
      <Card title="性能指標">
        <MetricGrid>
          <MetricCard
            label="平均檢索時間"
            value={`${memoryData.performance.avg_retrieval_time_ms}ms`}
            trend="down"
            trendValue="-5%"
          />
          <MetricCard
            label="緩存命中率"
            value={`${(memoryData.performance.cache_hit_rate * 100).toFixed(1)}%`}
            trend="up"
            trendValue="+3%"
          />
          <MetricCard
            label="同步速率"
            value={`${memoryData.performance.sync_operations_per_sec}/s`}
            trend="stable"
          />
        </MetricGrid>
      </Card>
      
      {/* 最近記憶列表 */}
      <Card title="最近記憶">
        <MemoryList memories={memoryData.recent_memories} />
      </Card>
    </div>
  );
}
